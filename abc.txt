        document.getElementById('gymForm').addEventListener('submit', function(event) {
            event.preventDefault();
            
            // Check if a time slot is selected
            const timeSelected = document.querySelector('input[name="time"]:checked');
            
            if (!timeSelected) {
                alert('Please select a preferred time slot.');
                return;
            }
            
            // If time slot is selected, show confirmation
            document.getElementById('confirmation').style.display = 'block';
            
            // Optional: Scroll to confirmation message
            document.getElementById('confirmation').scrollIntoView({ behavior: 'smooth' });
